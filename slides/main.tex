\begin{frame}[fragile]
    \frametitle{Phase 3: Julia --- rise of the JIT compilers}
    
    \begin{minted}{julia}

function quad(x0, α, n)
    x = x0
    for i in 1:(n-1)
        x = α * x * (1 - x)
    end
    return x
end

quad(0.2, 4.0, 10_000_000)
    \end{minted}

\end{frame}


\begin{frame}[fragile]
    \frametitle{Phase 3 continued: Python + Numba copy Julia}
    
    \begin{minted}{python}
from numba import jit

@jit
def quad(x0, α, n):
    x = x0
    for i in range(n-1):
        x = α * x * (1 - x)
    return x

quad(0.2, 4.0, 10_000_000)
    \end{minted}

\end{frame}
